#include <iostream>
#include <string>

using namespace std;

void countHanNum(int num, int* count)
{
	int first, second, diff, prev = 10;
	int numSize = to_string(num).size();
	int nCount = 0;
	
	if (num > 99)
	{
		int n = num;

		while (1)
		{
			if (n / 10 == 0) break;
			
			first = n % 10; 
			n = n / 10; 
			second = n % 10;
			diff = second - first;
			if (prev == diff) nCount++;
			prev = diff;

		}
		// cout << "numSize: " << numSize << endl;
		// cout << "nCount: " << nCount << endl;
		if (numSize - 2 == nCount) (*count)++;

	}
	else *count += 1;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int n;
	cin >> n;
	
	int count = 0;
	for (int i = 1; i <= n; i++)
	{
		countHanNum(i, &count);
	}

	cout << count << endl;

	return 0;
}
