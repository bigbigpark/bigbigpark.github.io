~~~ c++

#include <iostream>	
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int numCase;
	cin >> numCase;
	cin.ignore();
	
	int count = 0;

	for (int i = 0; i < numCase; i++)
	{
		string str;
		getline(cin, str);
		string temp = str;
		//aabbcc

		// temp: aabb
		// str:  ##bb

		bool isGroup = true;

		// aa
		int idx = 0;
		int prevIdx = 0;
		int j = 0;
		while(j < str.size())
		{
			while (1)
			{
				//cout << "temp:" << temp << endl;
				//cout << "str: " << str << endl;
				idx = str.find(temp[j]);
				//cout << "idx: " << idx << endl;

				if (idx == string::npos) break;

				str.replace(idx, 1, "#");

				//cout << "diff:" << prevIdx - idx << endl;
				if (idx - prevIdx > 1)
				{
					//cout << "$$$$$$$$$$$$" << endl;
					isGroup = false;
					break;
				}
				prevIdx = idx;
				//cout << "---" << endl;
			}
			if (!isGroup) break;

			//cout << "(bf)j : " << j << endl;
			j++;
			//cout << "(af)j : " << j << endl;
		}

		if (isGroup) count++;
	}

	cout << count << endl;

	return 0;
}
~~~
