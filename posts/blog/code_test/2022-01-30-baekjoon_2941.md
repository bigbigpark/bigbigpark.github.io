~~~c++

#include <iostream>	
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	vector<string> strArr = { "c=","c-","dz=","d-","lj","nj","s=","z=" };

	//ljes=njak
	string str;
	getline(cin, str);

	for (auto& it : strArr)
	{
		while (1)
		{
			int idx = str.find(it);

			if (idx == string::npos) break;

			str.replace(idx, it.size(), "#");
			//cout << "str: " << str << endl;
		}
	}
	cout << str.size() << endl;

	return 0;
}
~~~
