~~~c++
#include <iostream>
#include <vector>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	
	int n = 1000;

	vector<int> n_arr(n);

	int num = 0, sum = 0;
	for (int i = 0; i < n; i++)
	{
		sum = num = i + 1;

		while (1)
		{
			if (num == 0) break;
			sum += num % 10;
			num /= 10;
		}

		if (sum < n)
			n_arr[sum] = 1;
	}

	for (int i = 1; i < n_arr.size(); i++)
	{
		if (!n_arr[i]) cout << i << "\n";
	}
}
~~~
